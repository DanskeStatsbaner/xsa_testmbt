DO BEGIN
    DECLARE MYCOND_PASS CONDITION FOR SQL_ERROR_CODE 10001;
    DECLARE MYCOND_FAIL CONDITION FOR SQL_ERROR_CODE 10002;
	DECLARE EXIT HANDLER FOR SQLEXCEPTION 
		SELECT 'TESTTSTP00' as "SQL source", CASE WHEN ::SQL_ERROR_CODE = 10001 THEN 0 ELSE ::SQL_ERROR_CODE END as "Return code", ::SQL_ERROR_MESSAGE as "Message" FROM DUMMY;

	tab1 = SELECT 1 FROM DUMMY;
	if RECORD_COUNT( :tab1 ) > 0 then
		SIGNAL MYCOND_PASS SET MESSAGE_TEXT = 'OK';
	else
	    SIGNAL MYCOND_FAIL SET MESSAGE_TEXT = 'Not OK';
    end if;	

END;
